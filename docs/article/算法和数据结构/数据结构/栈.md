      this.queue1.push(element);

    if (this.queue1.length === 0 && this.queue2.length === 0) {
      return null;
    }

      nonEmptyQueue = this.queue2;
      emptyQueue = this.queue1;

      nonEmptyQueue = this.queue1;
      emptyQueue = this.queue2;

      emptyQueue.push(nonEmptyQueue.shift());
    }

  }
